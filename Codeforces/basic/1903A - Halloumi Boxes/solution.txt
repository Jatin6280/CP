#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    for (int i = 0; i < n; i++) {
        // Inside your loop
        int n, k;
        cin >> n >> k;
        vector<int> a(n);
        for(int i = 0; i < n; i++) cin >> a[i];

        if (k >= 2) {
            cout << "YES" << endl;
        } else {
            // Check if sorted
            bool sorted = true;
            for(int i = 0; i < n - 1; i++) {
                if(a[i] > a[i+1]) {
                    sorted = false;
                    break;
                }
            }
            if (sorted) cout << "YES" << endl;
            else cout << "NO" << endl;
        }
    }

    return 0;
}
