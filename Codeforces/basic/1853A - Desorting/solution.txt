#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> a(n);
        int min_diff = 1e9; // Initialize with a large value
        bool already_unsorted = false;

        for (int i = 0; i < n; i++) {
            cin >> a[i];
            if (i > 0) {
                // If it's already unsorted, we don't need any operations
                if (a[i] < a[i - 1]) {
                    already_unsorted = true;
                }
                // Calculate the gap between adjacent elements
                min_diff = min(min_diff, a[i] - a[i - 1]);
            }
        }

        if (already_unsorted) {
            cout << 0 << endl;
        } else {
            // The formula to break the sorted order for the smallest gap
            cout << (min_diff / 2) + 1 << endl;
        }
    }
    return 0;
}
