 #include <iostream>
#include <algorithm>
#include <cmath> // Required for abs()

using namespace std;

int main() {
    // Fast I/O for competitive programming
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    // Initialize with a value larger than any possible |Ai|
    int min_ops = 200000;

    for (int i = 0; i < n; ++i) {
        int a;
        cin >> a;
        // The cost to turn any number 'a' into 0 is abs(a).
        // We track the minimum cost found so far.
        min_ops = min(min_ops, abs(a));
    }

    cout << min_ops << endl;

    return 0;
}
