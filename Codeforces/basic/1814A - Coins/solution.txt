#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        long long n, k; // Using long long is safer for CP
        cin >> n >> k;

        // Condition 1: If n is even, we can always make it with 2s.
        // Condition 2: If n is odd, we subtract k.
        // If (n-k) is even and non-negative, the rest can be made with 2s.
        if (n % 2 == 0 || (n - k >= 0 && (n - k) % 2 == 0)) {
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }
    }
    return 0;
}
