#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;

        vector<int> v(n);

        int k = 0;
        for (int i = 0; i < n; i++) {
            cin >> v[i];
            if (v[i] == 2)
                k++;
        }

        if (k % 2 == 1) {
            cout << -1 << endl;
        }
        else if (k == 0) {
            cout << 1 << endl;
        }
        else {
            int cnt = 0;
            for (int i = 0; i < n; i++) {
                if (v[i] == 2)
                    cnt++;

                if (cnt == k / 2) {
                    cout << i + 1 << endl;
                    break;
                }
            }
        }
    }

    return 0;
}
