 #include <bits/stdc++.h>
using namespace std;

void solve() {
    int n, m;
    cin >> n >> m;
    string x, s;
    cin >> x >> s;

    int ops = 0;
    bool found = false;

    // Check up to 5-6 times (enough to make x longer than s)
    for (int i = 0; i <= 5; i++) {
        if (x.find(s) != string::npos) {
            found = true;
            cout << ops << "\n";
            return;
        }
        x += x; // Double x
        ops++;
    }

    cout << -1 << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
