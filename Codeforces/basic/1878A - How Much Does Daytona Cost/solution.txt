 #include <bits/stdc++.h>
using namespace std;
#include <vector>
bool toRepeat() {
    int n, k;
    cin >> n >> k;
    bool found = false;
    vector<int> v(n);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
        if (v[i] == k) {
            found = true;

        }
    }

    return found;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    cin>>t;
    while (t--) {
        if (toRepeat()) {
            cout<<"YES\n";
        }
        else {
            cout<<"NO\n";
        }
    }

}
